DROP TABLE IF EXISTS Setting;
DROP TABLE IF EXISTS ShiftBuffer;
CREATE TABLE IF NOT EXISTS Setting (
	User_ID  INTEGER  NOT NULL PRIMARY KEY   AUTOINCREMENT,
	Email VARCHAR (50) NOT NULL UNIQUE ,
    User_Password VARCHAR (25) NOT NULL,
    First_Name VARCHAR (25) NOT NULL,
    Last_Name VARCHAR (25) NOT NULL,
    User_TZ INTEGER NOT NULL ,
    User_Phone_Number VARCHAR (15) NOT NULL,
    User_active INTEGER  NOT NULL,
    User_Company_code VARCHAR (25) NOT NULL,
    User_Company_active INTEGER  NOT NULL
);
CREATE TABLE IF NOT EXISTS ShiftBuffer (
    ShiftBuffer_ID  INTEGER  NOT NULL PRIMARY KEY   AUTOINCREMENT,
    Shift_status  INTEGER  NOT NULL,
    shift_start_time DATETIME,
    shift_end_time DATETIME,
    shift_enter_lng REAL,
    shift_enter_lat REAL,
    shift_exit_lng REAL,
    shift_exit_lat REAL,
    SystemID String
);
